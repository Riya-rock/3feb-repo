Jenkins: pipeline


Step 1: Create a GitHub Repo
Create a repo named devops-pipeline-demo
 Add a file called Jenkinsfile with this content:

pipeline {

  agent any


  stages {

    stage('Build') {

      steps {

        echo 'Step 1: Building the application...'

      }

    }

    stage('Test') {

      steps {

        echo 'Step 2: Running tests...'

      }

    }

    stage('Deploy') {

      steps {

        echo 'Step 3: Deploying to staging...'

      }

    }

  }

}
